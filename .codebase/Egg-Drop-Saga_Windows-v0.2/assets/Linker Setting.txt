# Code::Blocks setting for windows: 

1. Change search directories to SDL2 -> C:\Libraries\SDL2\include\SDL2

2. Change Linker Settings -> 
-static -static-libgcc -static-libstdc++ -lmingw32 -lSDL2main -lSDL2 -lSDL2_mixer

3. Change Execute directory -> Project Directory

4. Link Libraries:
	mingw32
	SDL2_main
	SDL2
	SDL2_mixer

5. Icon setup for windows in main.cpp : 
	//Icon setup
    HWND hwnd = GetActiveWindow();
    HICON hIcon = (HICON)LoadImage(
        GetModuleHandle(NULL),
        MAKEINTRESOURCE(IDI_ICON1),
        IMAGE_ICON,
        32, 32,
        LR_DEFAULTCOLOR
    );
    SendMessage(hwnd, WM_SETICON, ICON_BIG, (LPARAM)hIcon);
    SendMessage(hwnd, WM_SETICON, ICON_SMALL, (LPARAM)hIcon);
